# deploy #
go to deploy section, set the new remote in cli.
put this in server.js
 // Static files for prod
if(process.env.NODE_ENV === 'production'){
    app.use(express.static('client/build'));

    app.get('*',(req,res) => {
        res.sendFile(path.resolve(__dirname,'client','build','index.html'));
    })
}

go to your app cd/client and npm run buid -> it compiles our fronend react app to a folder called build
if you dont want to build before every push to heroku you can write this scrip on the server side package.json -> "heroku-postbuild": "NPM_CONFIG_PRODUCTION=false npm install --prefix client && npm run build --prefix client"
